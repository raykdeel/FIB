82 double Cjt_Especies::species_distance(string id1, string id2){
83  if (id1 == id2) return 0;
84  map<string,Especie>::iterator it = mesp.find(id1);
85  Especie a = it->second;
86  it = mesp.find(id2);
87  Especie b = it->second;
88  map<string,int> k1;
89  map<string,int> k2;
90  map<string,int>::const_iterator it1;
91  map<string,int>::const_iterator it2;
92  k1 = a.query_kmer();
93  k2 = b.query_kmer();
94  it1 = k1.begin(), it2 = k2.begin();
95  double ig = 0, total = 0;
96  while (it1 != k1.end() and it2 != k2.end()) {
97  if (it1->first < it2->first) {
98  total += it1->second;
99  ++it1;
100  }
101  else if (it2->first < it1->first) {
102  total += it2->second;
103  ++it2;
104  }
105  else {
106  ig += min(it1->second, it2->second);
107  total += it2->second;
108  total += it1->second;
109  ++it1;
110  ++it2;
111  }
112  }
113  while (it1 != k1.end()) {
114  total += it1->second;
115  ++it1;
116  }
117  while (it2 != k2.end()) {
118  total += it2->second;
119  ++it2;
120  }
121  return (1 - (ig / (total-ig))) * 100;
122 }
